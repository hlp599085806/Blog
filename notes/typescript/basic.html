    TypeScript基本类型

    # TypeScript基本类型

    # 布尔类型

    //布尔类型 使用boolean来表示布尔类型 Boolean代表的是布尔对象

const isLoading: boolean = false;

    # 数字类型

    //数字类型 使用number来表示数字类型

//2进制 10进制 十六进制等 都可以用number类型来表示

const isNumber: number = 2;

const hexLiteral: number = 0xf00d;

const binaryLiteral: number = 0b1010;

const octalLiteral: number = 0o744;

    # 字符串类型

    //字符串类型 使用string来表示字符串类型

const isString: string = 'study TypeScript';

    # 空值

    //空值 当一个函数没有返回值的时候, 返回值的值类型为void

//不能将null赋值给void类型

const isVoid: void = undefined; //Good

const isVoid: void = null; //Bad

null类型和undefined类型

//null和undefined类型 它们有个各自的类型 作用不大

//默认情况下 null和undefined是所有类型的子类型

//开启--strictNullChecks后, null和undefined只能赋值给any和他们各自,
undefined可以赋值给void是例外

const isNull: null = null;

const isUndefined: undefined = undefined;

//不开启--strictNullChecks

const isUndefined: number = undefined;

    # Symbol类型

    //Symbol类型 需要再tsconfig.json里面lib配置es6的编译辅助库,

//它与其他类型不同 它是通过Symbol构造函数来使用的

const isSymbol = Symbol('Symbol1');

    # BigInt类型

    //BigInt类型 使用BigInt存储和大整数 使用BigInt的时候需要配置lib配置
ESNext的编译辅助库

const max = BigInt(Number.MAX_SAFE_INTEGER);

const max1 = max + BigInt(1);

const max2 = max + BigInt(2);
