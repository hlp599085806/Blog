    TypeScript其他常见类型

    # TypeScript其他常见类型

    # Any类型

    //想给一些编程阶段还不清除类型的变量

//可能来自于动态的内容和第三方的库

//这样会直接让它们通过编译阶段的检查

let isAny:any = 1;

isAny = 'changeString';

    # unknown类型

    //unknown是ts3.0引入的新类型，是any类型对应的安全类型

//unknown和any的主要区别是unknown类型会更加严格，

//unknown在执行大多数操作之前，会进行检查，而对any类型的值操作之前，不会进行任何检查

let value: unknown;

value = true; // OK

value = 1; // OK

value = "Hello World"; // OK

value = Symbol("type"); // OK

value = {} // OK

value = [] // OK

let value1: unknown;

value1.foo.bar; // ERROR

value1(); // ERROR

new value1(); // ERROR

value1[0][1]; // ERROR

//当unknown类型被确定是某个类型之前，它不能被进行任何操作，

//比如实例化，getter、函数执行等等

    # never类型

    //never类型表示永远也不会存在的值

//never类型是任何类型的子类型，可以赋值给任何类型

//没有类型是never的子类型或者赋值给never类型

//!!!any也不可以赋值给never

// 抛出异常的函数永远不会有返回值

function error(message: string): never {

throw new Error(message);

}

// 空数组，而且永远是空的

const empty: never[] = []

    # 数组类型

    //1.利用泛型去创建数组

const list: Array\<number\> = [1,2,3];

//2.常见数组(元素类型后面接上[])

const arr: number[] = [1,2,3];

    # 元组类型

    //元组是非常接近数组类型的，表示一个已知元素数量和类型的数组，各元素类型不必相同

let x:[string,number];

x = ['string',1]; //OK

x=['String']; //Error

x=['String',1,false];//Error

//元素的类型不对，或者元素的数量不对会报错

//可以把元组堪称严格版数组，比如[string,number]我们可以看成是

interface Tuple extends Array\<string \| number\> {

0: string;

1: number;

length: 2;

}

//元组越界问题

const tuple: [string, number] = ['a', 1];

tuple.push(2); // ok

console.log(tuple); // ["a", 1, 2] -\> 正常打印出来

//访问元素

console.log(tuple[2]); // Tuple type '[string, number]' of length '2' has no
element at index '2'
